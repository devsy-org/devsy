> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the app shell layout with sidebar navigation, theme switching (light/dark/system), and SvelteKit file-based routing for all views.

**Architecture:** A root `+layout.svelte` renders the sidebar + main content area. The sidebar uses shadcn navigation components. Theme is controlled via a CSS class on `<html>` and persisted to Tauri store. All stores are initialized in the root layout on mount.

**Tech Stack:** SvelteKit, shadcn-svelte (Button, Separator, Badge), Tailwind CSS v4

### Task 3: Create theme store and switcher

**Files:**
- Create: `desktop-new/src/lib/stores/settings.ts`
- Create: `desktop-new/src/lib/components/layout/ThemeSwitcher.svelte`

- [ ] **Step 1: Create lib/stores/settings.ts**

```ts
import { writable } from "svelte/store";

export type Theme = "light" | "dark" | "system";

function createSettingsStore() {
  const { subscribe, set, update } = writable<{
    theme: Theme;
  }>({
    theme: "system",
  });

  return {
    subscribe,
    setTheme(theme: Theme) {
      update((s) => ({ ...s, theme }));
      applyTheme(theme);
      // Persist to localStorage (Tauri store can be added later)
      try {
        localStorage.setItem("devpod-theme", theme);
      } catch {}
    },
    init() {
      try {
        const saved = localStorage.getItem("devpod-theme") as Theme | null;
        if (saved) {
          set({ theme: saved });
          applyTheme(saved);
        } else {
          applyTheme("system");
        }
      } catch {
        applyTheme("system");
      }
    },
  };
}

function applyTheme(theme: Theme) {
  if (typeof document === "undefined") return;

  const root = document.documentElement;
  root.classList.remove("light", "dark");

  if (theme === "system") {
    const prefersDark = window.matchMedia(
      "(prefers-color-scheme: dark)",
    ).matches;
    root.classList.add(prefersDark ? "dark" : "light");
  } else {
    root.classList.add(theme);
  }
}

export const settings = createSettingsStore();
```

- [ ] **Step 2: Create ThemeSwitcher.svelte**

```svelte
<script lang="ts">
  import { Button } from "$lib/components/ui/button";
  import { settings, type Theme } from "$lib/stores/settings";

  let currentTheme = $state<Theme>("system");

  settings.subscribe((s) => {
    currentTheme = s.theme;
  });

  const themes: { value: Theme; label: string }[] = [
    { value: "light", label: "Light" },
    { value: "dark", label: "Dark" },
    { value: "system", label: "System" },
  ];

  function cycle() {
    const idx = themes.findIndex((t) => t.value === currentTheme);
    const next = themes[(idx + 1) % themes.length];
    settings.setTheme(next.value);
  }
</script>

<Button variant="ghost" size="sm" on:click={cycle} class="text-xs text-muted-foreground">
  {themes.find((t) => t.value === currentTheme)?.label ?? "System"}
</Button>
```

  import "../app.css";
  import Sidebar from "$lib/components/layout/Sidebar.svelte";
  import ThemeSwitcher from "$lib/components/layout/ThemeSwitcher.svelte";
  import { workspaces } from "$lib/stores/workspaces";
  import { providers } from "$lib/stores/providers";
  import { machines } from "$lib/stores/machines";
  import { settings } from "$lib/stores/settings";

  onMount(() => {
    settings.init();
    workspaces.init();
    providers.init();
    machines.init();

  onDestroy(() => {
    workspaces.destroy();
    providers.destroy();
    machines.destroy();
  });
</script>

  <Sidebar />

  <div class="flex flex-1 flex-col overflow-hidden">
    <!-- Top bar -->
    <header class="flex h-14 items-center justify-end border-b border-border px-4">
      <ThemeSwitcher />
    </header>

    <!-- Main content -->
    <main class="flex-1 overflow-auto p-6">
      {@render children()}
    </main>
  </div>
</div>
